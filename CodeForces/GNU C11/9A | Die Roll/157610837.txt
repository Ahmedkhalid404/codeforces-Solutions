#include<stdio.h>
int main(){
    int x,y,max;
    scanf("%d%d",&x,&y);
    max=x;
    if(y>x)
        max=y;
    if(max==6)
        printf("1/6");
    else if(max==5)
        printf("1/3");
    else if(max==4)
        printf("1/2");
    else if(max==3)
        printf("2/3");
    else if(max==2)
        printf("5/6");
    else if(max==1)
        printf("1/1");
return 0;
}
